package com.hr.backend.common; // 声明包名

@Data // 自动生成getter/setter/toString等方法
public class Response<T> { // 定义泛型响应类
    private int code; // 响应码
    private String message; // 响应消息

        r.setMessage("success"); // 设置响应消息
        r.setData(data); // 设置响应数据
        return r; // 返回响应对象
    }

    public static <T> Response<T> error(int code, String message) { // 静态方法：构建错误响应
        Response<T> r = new Response<>(); // 创建响应对象
        r.setCode(code); // 设置错误码
        r.setMessage(message); // 设置错误消息
        r.setData(null); // 设置数据为null
        return r; // 返回响应对象

#### ✅ ReportController.java - 报表控制器
**文件路径：** `backend/src/main/java/com/hr/backend/controller/ReportController.java`

**逐行注释示例：**
```java
package com.hr.backend.controller; // 声明包名
import com.hr.backend.common.Response; // 导入响应类
import com.hr.backend.model.entity.ReportTemplate; // 导入报表模板实体
import com.hr.backend.service.ReportTemplateService; // 导入报表模板服务
import com.hr.backend.service.SqlValidationService; // 导入SQL校验服务
import org.springframework.jdbc.core.JdbcTemplate; // 导入JDBC模板
import org.springframework.web.bind.annotation.*; // 导入Web注解

@RestController // 标记为REST控制器
@RequestMapping("/api/report") // 设置基础路径
public class ReportController { // 定义控制器类

    @Resource // 依赖注入
    private ReportTemplateService reportTemplateService; // 注入报表模板服务

    @Resource // 依赖注入
    private SqlValidationService sqlValidationService; // 注入SQL校验服务

    @Resource // 依赖注入
    private JdbcTemplate jdbcTemplate; // 注入JDBC模板

    @GetMapping("/list") // 映射GET请求
    public Response<List<ReportTemplate>> list() { // 获取报表列表方法
        try { // 开始try块，捕获异常
            return Response.success(reportTemplateService.list()); // 返回报表列表
        } catch (Exception e) { // 捕获异常
            return Response.error(500, "获取报表列表失败：" + e.getMessage()); // 返回错误响应
        }
    }

    @GetMapping("/{id}") // 映射GET请求，带路径参数
    public Response<ReportTemplate> getById(@PathVariable Long id) { // 根据ID获取报表方法
        try { // 开始try块，捕获异常
            ReportTemplate template = reportTemplateService.getById(id); // 查询报表
            if (template == null) { // 检查报表是否存在
                return Response.error(404, "报表模板不存在"); // 返回404错误
            }
            return Response.success(template); // 返回报表信息
        } catch (Exception e) { // 捕获异常
            return Response.error(500, "获取报表详情失败：" + e.getMessage()); // 返回错误响应
        }
    }

    @PostMapping // 映射POST请求
    public Response<Void> add(@RequestBody ReportTemplate template) { // 新增报表方法
        try { // 开始try块，捕获异常
            if (template.getQuerySql() != null && !template.getQuerySql().trim().isEmpty()) { // 检查SQL是否为空
                Map<String, Object> validationResult = sqlValidationService.validateSql(template.getQuerySql()); // 校验SQL
                if (!(Boolean) validationResult.get("valid")) { // 检查校验结果
                    return Response.error(400, "SQL校验失败：" + validationResult.get("message")); // 返回400错误
                }
            }
            reportTemplateService.save(template); // 保存报表
            return Response.success(); // 返回成功响应
        } catch (Exception e) { // 捕获异常
            return Response.error(500, "新增报表失败：" + e.getMessage()); // 返回错误响应
        }
    }

    @PostMapping("/preview") // 映射POST请求
    public Response<Map<String, Object>> preview(@RequestBody Map<String, String> request) { // 预览SQL方法
        try { // 开始try块，捕获异常
            String sql = request.get("sql"); // 获取SQL参数

            if (sql == null || sql.trim().isEmpty()) { // 检查SQL是否为空
                return Response.error(400, "SQL语句不能为空"); // 返回400错误
            }

            Map<String, Object> validationResult = sqlValidationService.validateSql(sql); // 校验SQL
            if (!(Boolean) validationResult.get("valid")) { // 检查校验结果
                return Response.error(400, "SQL校验失败：" + validationResult.get("message")); // 返回400错误
            }

            String sanitizedSql = sqlValidationService.sanitizeSql(sql); // 清理SQL

            String limitedSql = sanitizedSql.replaceAll("(?i)LIMIT\\s+\\d+", "LIMIT 10"); // 替换LIMIT
            if (!limitedSql.matches("(?i).*LIMIT.*")) { // 检查是否包含LIMIT
                limitedSql += " LIMIT 10"; // 添加LIMIT
            }

            List<Map<String, Object>> data = jdbcTemplate.queryForList(limitedSql); // 执行查询

            Set<String> columns = new LinkedHashSet<>(); // 创建列名集合
            if (!data.isEmpty()) { // 检查是否有数据
                columns.addAll(data.get(0).keySet()); // 获取列名
            }

            Map<String, Object> result = new HashMap<>(); // 创建结果Map
            result.put("data", data); // 添加数据
            result.put("columns", columns); // 添加列名
            result.put("count", data.size()); // 添加数量

            return Response.success(result); // 返回成功响应
        } catch (Exception e) { // 捕获异常
            return Response.error(500, "SQL执行失败：" + e.getMessage()); // 返回错误响应
        }
    }
}
```

package com.hr.backend.service.impl; // 声明包名
import com.baomidou.mybatisplus.core.conditions.query.LambdaQueryWrapper; // 导入MyBatis Plus查询条件
import com.hr.backend.common.BaseException; // 导入业务异常类
import com.hr.backend.common.ErrorCode; // 导入错误码枚举
import com.hr.backend.mapper.UserMapper; // 导入用户Mapper
import com.hr.backend.model.dto.LoginRequest; // 导入登录请求DTO
import com.hr.backend.model.dto.LoginResponse; // 导入登录响应DTO
import com.hr.backend.model.entity.User; // 导入用户实体
import com.hr.backend.security.JwtUtil; // 导入JWT工具类
import com.hr.backend.service.UserService; // 导入用户服务接口
import org.springframework.security.crypto.password.PasswordEncoder; // 导入密码编码器
import org.springframework.stereotype.Service; // 导入服务注解
import javax.annotation.Resource; // 导入依赖注入注解

@Service // 标记为服务类
public class UserServiceImpl extends ServiceImpl<UserMapper, User> implements UserService { // 定义服务实现类

    @Resource // 依赖注入
    private PasswordEncoder passwordEncoder; // 注入密码编码器

    @Resource // 依赖注入
    private JwtUtil jwtUtil; // 注入JWT工具类

    @Override // 重写接口方法
    public LoginResponse login(LoginRequest request) { // 登录方法
        User user = baseMapper.selectByUsername(request.getUsername()); // 根据用户名查询用户
        if (user == null) { // 检查用户是否存在
            throw new BaseException(ErrorCode.UNAUTHORIZED.getCode(), "用户名或密码错误"); // 抛出未授权异常
        }
        if (user.getStatus() != null && user.getStatus() != 1) { // 检查用户状态
            throw new BaseException(ErrorCode.FORBIDDEN.getCode(), "账号已禁用"); // 抛出权限不足异常
        }
        if (!passwordEncoder.matches(request.getPassword(), user.getPassword())) { // 验证密码
            throw new BaseException(ErrorCode.UNAUTHORIZED.getCode(), "用户名或密码错误"); // 抛出未授权异常
        }
        String token = jwtUtil.generateToken(user.getId(), user.getUsername(), user.getRole()); // 生成JWT令牌
        return LoginResponse.of(token, user); // 返回登录响应
    }

    @Override // 重写接口方法
    public void register(RegisterRequest request) { // 注册方法
        if (baseMapper.selectByUsername(request.getUsername()) != null) { // 检查用户名是否存在
            throw new BaseException(ErrorCode.USER_EXISTS.getCode(), "工号已存在"); // 抛出用户已存在异常
        }
        User user = new User(); // 创建用户对象
        user.setUsername(request.getUsername()); // 设置用户名
        user.setPassword(passwordEncoder.encode(request.getPassword())); // 加密密码
        user.setName(request.getName()); // 设置姓名
        user.setRole(request.getRole()); // 设置角色
        user.setDeptId(request.getDeptId()); // 设置部门ID
        user.setDeptName(request.getDeptName()); // 设置部门名称
        user.setDeptScope(request.getDeptScope()); // 设置部门范围
        user.setPhone(request.getPhone()); // 设置手机号
        user.setEmail(request.getEmail()); // 设置邮箱
        user.setStatus(1); // 设置状态为启用
        save(user); // 保存用户到数据库
    }

    @Override // 重写接口方法
    public void changePassword(ChangePasswordRequest request) { // 修改密码方法
        User current = SecurityUtil.getCurrentUser(); // 获取当前用户
        if (current == null) { // 检查是否登录
            throw new BaseException(ErrorCode.UNAUTHORIZED.getCode(), "请先登录"); // 抛出未授权异常
        }
        User user = getById(current.getId()); // 根据ID查询用户
        if (user == null || !passwordEncoder.matches(request.getOldPassword(), user.getPassword())) { // 验证原密码
            throw new BaseException(ErrorCode.PARAM_ERROR.getCode(), "原密码错误"); // 抛出参数错误异常
        }
        user.setPassword(passwordEncoder.encode(request.getNewPassword())); // 加密新密码
        updateById(user); // 更新用户信息
    }
}

     * @return JWT令牌字符串
     */
    public String generateToken(Long userId, String username, String role) { // 生成令牌方法
        Map<String, Object> claims = new HashMap<>(); // 创建Claims Map
        claims.put("userId", userId); // 添加用户ID
        claims.put("username", username); // 添加用户名
        claims.put("role", role); // 添加角色

        Date now = new Date(); // 获取当前时间
        Date expirationDate = new Date(now.getTime() + EXPIRATION_TIME); // 计算过期时间

        return Jwts.builder() // 创建JWT构建器
                .setClaims(claims) // 设置Claims
                .setSubject(username) // 设置主题
                .setIssuedAt(now) // 设置签发时间
                .setExpiration(expirationDate) // 设置过期时间
                .signWith(SignatureAlgorithm.HS256, SECRET_KEY) // 使用HS256算法签名
                .compact(); // 构建JWT字符串
    }

    /**
     * 解析JWT令牌
     * @param token JWT令牌字符串
     * @return Claims对象
     */
    public Claims parseToken(String token) { // 解析令牌方法
        return Jwts.parser() // 创建JWT解析器
                .setSigningKey(SECRET_KEY) // 设置签名密钥
                .parseClaimsJws(token) // 解析JWT
                .getBody(); // 获取Claims
    }

    /**
     * 从令牌中获取用户ID
     * @param token JWT令牌字符串
     * @return 用户ID
     */
    public Long getUserIdFromToken(String token) { // 获取用户ID方法
        Claims claims = parseToken(token); // 解析令牌
        return Long.valueOf(claims.get("userId").toString()); // 从Claims中获取用户ID
    }

    /**
     * 从令牌中获取用户名
     * @param token JWT令牌字符串
     * @return 用户名
     */
    public String getUsernameFromToken(String token) { // 获取用户名方法
        Claims claims = parseToken(token); // 解析令牌
        return claims.getSubject(); // 从Claims中获取主题（用户名）
    }

    /**
     * 验证令牌是否过期
     * @param token JWT令牌字符串
     * @return true表示已过期，false表示未过期
     */
    public boolean isTokenExpired(String token) { // 验证令牌过期方法
        Claims claims = parseToken(token); // 解析令牌
        return claims.getExpiration().before(new Date()); // 检查过期时间是否早于当前时间
    }
}
